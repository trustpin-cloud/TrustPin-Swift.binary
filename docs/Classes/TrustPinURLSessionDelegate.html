TrustPinURLSessionDelegate

public final class TrustPinURLSessionDelegate : NSObject, URLSessionDelegate

A URLSessionDelegate that performs certificate pinning using a specific TrustPin instance.

Default instance

// Uses TrustPin.default — equivalent to pre-multi-instance behaviour
let delegate = TrustPinURLSessionDelegate()
let session = URLSession(configuration: .default, delegate: delegate, delegateQueue: nil)

Named instance (library use)

let pin = TrustPin.instance(id: "com.mylib")
try await pin.setup(config)

let delegate = pin.makeURLSessionDelegate()
// or equivalently:
let delegate = TrustPinURLSessionDelegate(instance: pin)
let session = URLSession(configuration: .default, delegate: delegate, delegateQueue: nil)
